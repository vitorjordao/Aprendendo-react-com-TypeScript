# Comandos para instalação in inicial

```bash
create-react-app tsreact --typescript
sudo npm install -g typescript
sudo yarn add eslint-plugin-jsx-a11y -D
sudo yarn add eslint-config-airbnb eslint-plugin-import eslint-plugin-react eslint-config-prettier eslint-plugin-prettier prettier @typescript-eslint/eslint-plugin @typescript-eslint/parser -D
sudo yarn add eslint-import-resolver-typescript -D
```
